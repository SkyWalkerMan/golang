    import "fmt"

        fmt.Println("**     **            **   **                 **     **")
        fmt.Println("** * **              *******                 *******")
        fmt.Println("** * *               *******                 * * ** ")
        fmt.Println("**                   *******                 * *")
        fmt.Println("**                   **   **                 * *")
        fmt.Println("**                   **   **                 * *")
        fmt.Println("**                   **   **                 * *")
        fmt.Println("**                   **   **                 * *")
    }
